64  UkladW(std::array<double, 3> sr, double kat_deg, char os) : srodek(sr) {
65  double kat_rad = kat_deg * 3.14159265 / 180.0;
66  double c = std::cos(kat_rad);
67  double s = std::sin(kat_rad);
68  for (int i = 0; i < 3; ++i)
69  for (int j = 0; j < 3; ++j) orientacja[i][j] = 0.0;
70  switch (os) {
71  case 'X': {
72  orientacja[0][0] = 1.0;
73  orientacja[1][1] = c;
74  orientacja[2][2] = c;
75  orientacja[1][2] = -s;
76  orientacja[2][1] = s;
77  break;
78  }
79  case 'Y': {
80  orientacja[1][1] = 1.0;
81  orientacja[0][0] = c;
82  orientacja[2][2] = c;
83  orientacja[2][0] = -s;
84  orientacja[0][2] = s;
85  break;
86  }
87  case 'Z': {
88  orientacja[2][2] = 1.0;
89  orientacja[0][0] = c;
90  orientacja[1][1] = c;
91  orientacja[0][1] = -s;
92  orientacja[1][0] = s;
93  break;
94  }
95  }

106  }
