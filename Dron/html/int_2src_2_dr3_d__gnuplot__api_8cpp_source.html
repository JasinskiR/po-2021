47 APIGnuPlot3D::APIGnuPlot3D(double minX, double maxX, double minY, double maxY,
48  double minZ, double maxZ, int ref_time_ms)
49  : Draw3DAPI(ref_time_ms), refresh_thread_ptr(nullptr), last_shape(0) {
50  pipe(pipe_fd);
51  int local_pid = fork();
52  if (local_pid != 0) {
53  gnuplot_pid = local_pid;
54  stringstream ss_tmp;
55 
56  ss_tmp << " set terminal qt" << endl
57  << "set xrange [" << minX << ":" << maxX << "]" << endl
58  << "set yrange [" << minY << ":" << maxY << "]" << endl
59  << "set zrange [" << minZ << ":" << maxZ << "]" << endl
60  << "set xlabel \"X\"" << endl
61  << "set ylabel \"Y\"" << endl
62  << "set zlabel \"Z\"" << endl
63  << "set cbrange [ -1 : 1 ] noreverse writeback" << endl
64  << "set rrange [ -1 : 1 ] noreverse writeback" << endl
65  << "set view equal xyz" << endl
66  << "set xyplane relative 0" << endl
67  << "unset key" << endl
68  << "set grid" << endl
69  << "NO_ANIMATION = 1" << endl
70  //<< "set palette model RGB rgbformulae 7,5,15" << endl
71  //<< "set pm3d at s " << endl
72  //<< "unset colorbox " << endl
73 
74  << "set hidden3d" << endl;
75 
76  intro = ss_tmp.str();
77  send2gnuplot(intro);
78  if (ref_time_ms > 0) {
79  refresh_thread_ptr.reset(new thread(&APIGnuPlot3D::replot_loop, this));
80  refresh_thread_ptr->detach();
81  }
82  } else {
83  dup2(pipe_fd[0], 0);

109 void APIGnuPlot3D::draw_all_shapes() {
110  stringstream ss;
111  string color;
112  ss << intro << "splot ";
113  for (auto itel = shapes.begin(); itel != shapes.end();) {
114  color = translate_color(itel->second.second);
115  if (!color.compare("error")) {
116  color = translate_color("black");
117  std::cerr << "Shape nr " << itel->first
118  << " has invalid color.\nUsing black color to draw it." << endl;
119  }
120  ss << "\'-\' using 1:2:3 with lines lt 1 lw 3 " << color;
121  if (++itel != shapes.end()) ss << ",";
122  }
123  ss << endl;
124  for (auto& el : shapes) {
125  for (auto& inel : el.second.first) {
126  for (auto& in2el : inel) {
127  ss << in2el;
128  }
129  ss << endl;
130  }
131  ss << "e" << endl;
132  }
133  ss << "pause -1" << endl;
134  command = ss.str();
135  send2gnuplot(command);
136 }

138 string APIGnuPlot3D::translate_color(const string& color) const {
139  string out = "lc rgb \'#";
140  if (!color.compare("black")) return out + "000000\'";
141  if (!color.compare("white")) return out + "ffffff\'";
142  if (!color.compare("grey")) return out + "808080\'";
143  if (!color.compare("blue")) return out + "0072bd\'";
144  if (!color.compare("orange")) return out + "d95319\'";
145  if (!color.compare("yellow")) return out + "edb120\'";
146  if (!color.compare("purple")) return out + "7e2f8e\'";
147  if (!color.compare("green")) return out + "77ac30\'";
148  if (!color.compare("light-blue")) return out + "4dbeee\'";
149  if (!color.compare("red")) return out + "a2142f\'";
150  if (!color.compare(0, 5, "rgb:#")) return out + color.substr(5);
151  return "error";
152 }
